### OpenAPI 3.0 Structure Template
```yaml
openapi: 3.0.0
info:
  title: Casino Proxy API
  version: 1.0.0
  description: |
    Unified gaming provider webhook gateway API
    Supporting 8 gaming providers...

servers:
  - url: https://api.casino-proxy.local/v1
    description: Production

paths:
  # Admin endpoints
  /entry: { ... }
  /status: { ... }
  /image: { ... }
  /internal/operator/: { ... }
  
  # Pragmatic Play
  /webhooks/pragmatic-play/{endpoint}: { ... }
  
  # Evolution Gaming
  /webhooks/evolution/authentication: { ... }
  /webhooks/evolution/debit: { ... }
  /webhooks/evolution/credit: { ... }
  /webhooks/evolution/rollback: { ... }
  /webhooks/evolution/getNewToken: { ... }
  
  # ... etc for other providers

components:
  securitySchemes:
    hmacSignature:
      type: apiKey
      name: X-Signature
      in: header
    
  schemas:
    Error: { ... }
    Transaction: { ... }
    # ... shared schemas
```
